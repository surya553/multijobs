pipeline {
    agent any
    parameters {
        choice(choices: ['angular', 'maven'],description: '', name: 'select the job for changing directory')
    }    
    stages {    
        stage("git checkout") {
            steps {    
                git url: 'https://github.com/surya553/multijobs.git'
            }
        }  
        stage("changing the directory"){ 
            steps { 
                dir("/var/lib/jenkins/workspace/${params.'select the job for changing directory'}") {
                    sh "pwd"
                } 
            }
        } 
        stage("loading jenkins file") {
            steps {
                load '/var/lib/jenkins/workspace/${params.'select the job for changing directory\'}/Jenkinsfile.groovy'
            }
        }    
    }
}
