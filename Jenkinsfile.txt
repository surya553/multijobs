pipeline {
    agent any
    parameters {
        choice(choices: ['angular', 'maven'],description: '', name: 'select the job for changing directory')
    }    
    stages {    
        stage("git checkout") {
            steps {    
                git url: 'https://github.com/surya553/multijobs.git'
            }
        }  
        stage("changing the directory"){ 
            steps { 
                dir("$pwd/workspace/${params.'select the job for changing directory'}") {
                    sh "pwd"
                } 
            }
        } 
        stage("loading jenkins file") {
            steps {
                validateDeclarativePipeline "$pwd()/Jenkinsfile.txt"
            }
        }    
    }
}
